package co.edu.uniminuto.secondactivity;

import android.content.Intent;
import android.os.Bundle;
import android.widget.Button;
import android.widget.EditText;
import androidx.appcompat.app.AppCompatActivity;

public class EditarTareaActivity extends AppCompatActivity {
    EditText editTextTarea;
    Button btnGuardar, btnEliminar;
    int posicion;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_editar_tarea);

        editTextTarea = findViewById(R.id.editTextTarea);
        btnGuardar = findViewById(R.id.btnGuardar);
        btnEliminar = findViewById(R.id.btnEliminar);

        Intent intent = getIntent();
        String tarea = intent.getStringExtra("tarea");
        posicion = intent.getIntExtra("posicion", -1);
        editTextTarea.setText(tarea);

        btnGuardar.setOnClickListener(v -> {
            String tareaModificada = editTextTarea.getText().toString();
            Intent resultIntent = new Intent();
            resultIntent.putExtra("accion", "editar");
            resultIntent.putExtra("tarea", tareaModificada);
            resultIntent.putExtra("posicion", posicion);
            setResult(RESULT_OK, resultIntent);
            finish();
        });

        btnEliminar.setOnClickListener(v -> {
            Intent resultIntent = new Intent();
            resultIntent.putExtra("accion", "eliminar");
            resultIntent.putExtra("posicion", posicion);
            setResult(RESULT_OK, resultIntent);
            finish();
        });
    }
}
